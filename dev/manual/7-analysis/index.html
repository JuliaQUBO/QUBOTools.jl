using Plots

L = Dict{Int,Float64}(1 => 0.5, 2 => 2.0, 3 => -3.0)
Q = Dict{Tuple{Int,Int},Float64}((1,2) => 2.0, (1,3) => -2.0, (2,3) => 0.5)

m = Model{Int,Float64,Int}(L, Q; domain=:bool)

plot(m)
